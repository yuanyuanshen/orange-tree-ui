# 安装

---

### Install

```js
npm i orange-tree-ui
```

### Quick Start

```js
import OrangeTreeUI from 'orange-tree-ui'
import 'orange-tree-ui/packages/theme-default/lib/index.css'

Vue.use(OrangeTreeUI)

// or
import {
  Button
  // ...
} from 'orange-tree-ui'

Vue.component(Button.name, Button)
```
