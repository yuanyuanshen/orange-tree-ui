# 安装

---

### Install

```js
npm i green-orange-ui
```

### Quick Start

```js
import GreenOrangeUI from 'green-orange-ui'
import 'green-orange-ui/packages/theme-default/lib/index.css'

Vue.use(GreenOrangeUI)

// or
import {
  Button
  // ...
} from 'green-orange-ui'

Vue.component(Button.name, Button)
```
